import json
import os

MEMORY_FILE = "memory.json"

# Memory load
if os.path.exists(MEMORY_FILE):
    with open(MEMORY_FILE, "r") as f:
        memory = json.load(f)
else:
    memory = {
        "hello": "Hi! Kaise ho?",
        "name": "Mera naam Myzeriyana hai.",
        "python": "Python ek powerful programming language hai."
    }


def save_memory():
    with open(MEMORY_FILE, "w") as f:
        json.dump(memory, f, indent=4)


def generate_reply(message):
    words = message.lower().split()

    replies = []

    for word in words:
        if word in memory:
            replies.append(memory[word])

    if replies:
        return " ".join(replies)

    else:
        return "Mujhe iska matlab nahi pata. Please sikhaiye."


def learn_word(word, meaning):
    memory[word] = meaning
    save_memory()
    print(f"✅ Maine '{word}' seekh liya!")


print("Myzeriyana Basic AI started 🤖")
print("Type 'exit' to stop")

while True:

    user_input = input("\nTum: ")

    if user_input.lower() == "exit":
        break

    reply = generate_reply(user_input)
    print("AI:", reply)

    if "nahi pata" in reply:
        teach = input("Iska matlab kya hota hai? ")
        learn_word(user_input.lower(), teach)

import json
import os

BRAIN_FILE = "brain.json"


# Load brain
def load_brain():
    if os.path.exists(BRAIN_FILE):
        with open(BRAIN_FILE, "r", encoding="utf-8") as f:
            return json.load(f)
    else:
        return {}


# Save brain
def save_brain(brain):
    with open(BRAIN_FILE, "w", encoding="utf-8") as f:
        json.dump(brain, f, indent=4)


brain = load_brain()


def reply(message):
    words = message.lower().split()

    answers = []

    for word in words:
        if word in brain:
            answers.append(brain[word])

    if answers:
        return " ".join(answers)

    return None


print("Myzeriyana started 🤖")
print("Type exit to stop\n")


while True:

    user = input("Tum: ").lower()

    if user == "exit":
        print("AI band ho rahi hai...")
        break

    response = reply(user)

    if response:
        print("AI:", response)

    else:
        print("AI: Mujhe iska matlab nahi pata. Sikhayenge?")

        meaning = input("Meaning likho: ")

        brain[user] = meaning

        save_brain(brain)

        print("✅ Maine seekh liya!")

import json
import os

DICT_FILE = "dictionary.txt"
BRAIN_FILE = "brain.json"


# ---------------- LOAD DICTIONARY ----------------
def load_dictionary():
    dictionary = {}

    if os.path.exists(DICT_FILE):
        with open(DICT_FILE, "r", encoding="utf-8") as f:
            for line in f:
                if "=" in line:
                    word, meaning = line.strip().split("=", 1)
                    dictionary[word.strip()] = meaning.strip()

    return dictionary


# ---------------- LOAD BRAIN ----------------
def load_brain():
    if os.path.exists(BRAIN_FILE):
        with open(BRAIN_FILE, "r", encoding="utf-8") as f:
            return json.load(f)
    return {}


# ---------------- SAVE BRAIN ----------------
def save_brain(brain):
    with open(BRAIN_FILE, "w", encoding="utf-8") as f:
        json.dump(brain, f, indent=4)


dictionary = load_dictionary()
brain = load_brain()


# ---------------- REPLY ENGINE ----------------
def generate_reply(message):

    words = message.lower().split()

    replies = []

    for word in words:

        if word in brain:
            replies.append(brain[word])

        elif word in dictionary:
            replies.append(dictionary[word])

        else:
            replies.append(f"{word} unknown")

    return " | ".join(replies)


# ---------------- LEARNING SYSTEM ----------------
def learn(word, meaning):

    brain[word] = meaning
    save_brain(brain)

    print("✅ Learned:", word)


print("Myzeriyana Dictionary AI started 🤖")
print("Type EXIT to stop\n")


while True:

    user = input("You: ").lower()

    if user == "exit":
        break

    words = user.split()

    unknown_words = [w for w in words if w not in dictionary and w not in brain]

    if unknown_words:

        for word in unknown_words:
            print(f"AI: '{word}' ka meaning batao:")
            meaning = input("Meaning: ")
            learn(word, meaning)

    else:

        reply = generate_reply(user)
        print("AI:", reply)
